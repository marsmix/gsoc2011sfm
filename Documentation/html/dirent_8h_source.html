00198 /*****************************************************************************
00199  * Open directory stream DIRNAME for read and return a pointer to the
00200  * internal working area that is used to retrieve individual directory
00201  * entries.
00202  */
00203 static DIR *opendir(const char *dirname)
00204 {
00205    DIR *dirp;
00206 
00207    /* ensure that the resulting search pattern will be a valid file name */
00208    if (dirname == NULL) {
00209       DIRENT_SET_ERRNO (ENOENT);
00210       return NULL;
00211    }
00212    if (strlen (dirname) + 3 >= MAX_PATH) {
00213       DIRENT_SET_ERRNO (ENAMETOOLONG);
00214       return NULL;
00215    }
00216 
00217    /* construct new DIR structure */
00218    dirp = (DIR*) malloc (sizeof (struct DIR));
00219    if (dirp != NULL) {
00220       int error;
00221 
00222       /*
00223        * Convert relative directory name to an absolute one.  This
00224        * allows rewinddir() to function correctly when the current working
00225        * directory is changed between opendir() and rewinddir().
00226        */
00227       if (GetFullPathNameA (dirname, MAX_PATH, dirp->patt, NULL)) {
00228          char *p;
00229 
00230          /* append the search pattern "\\*\0" to the directory name */
00231          p = strchr (dirp->patt, '\0');
00232          if (dirp->patt < p  &&  *(p-1) != '\\'  &&  *(p-1) != ':') {
00233            *p++ = '\\';
00234          }
00235          *p++ = '*';
00236          *p = '\0';
00237 
00238          /* open directory stream and retrieve the first entry */
00239          dirp->search_handle = FindFirstFileA (dirp->patt, &dirp->find_data);
00240          if (dirp->search_handle != INVALID_HANDLE_VALUE) {
00241             /* a directory entry is now waiting in memory */
00242             dirp->cached = 1;
00243             error = 0;
00244          } else {
00245             /* search pattern is not a directory name? */
00246             DIRENT_SET_ERRNO (ENOENT);
00247             error = 1;
00248          }
00249       } else {
00250          /* buffer too small */
00251          DIRENT_SET_ERRNO (ENOMEM);
00252          error = 1;
00253       }
00254 
00255       if (error) {
00256          free (dirp);
00257          dirp = NULL;
00258       }
00259    }
00260 
00261    return dirp;
00262 }

00265 /*****************************************************************************
00266  * Read a directory entry, and return a pointer to a dirent structure
00267  * containing the name of the entry in d_name field.  Individual directory
00268  * entries returned by this very function include regular files,
00269  * sub-directories, pseudo-directories "." and "..", but also volume labels,
00270  * hidden files and system files may be returned.
00271  */
00272 static struct dirent *readdir(DIR *dirp)
00273 {
00274    DWORD attr;
00275    if (dirp == NULL) {
00276       /* directory stream did not open */
00277       DIRENT_SET_ERRNO (EBADF);
00278       return NULL;
00279    }
00280 
00281    /* get next directory entry */
00282    if (dirp->cached != 0) {
00283       /* a valid directory entry already in memory */
00284       dirp->cached = 0;
00285    } else {
00286       /* get the next directory entry from stream */
00287       if (dirp->search_handle == INVALID_HANDLE_VALUE) {
00288          return NULL;
00289       }
00290       if (FindNextFileA (dirp->search_handle, &dirp->find_data) == FALSE) {
00291          /* the very last entry has been processed or an error occured */
00292          FindClose (dirp->search_handle);
00293          dirp->search_handle = INVALID_HANDLE_VALUE;
00294          return NULL;
00295       }
00296    }
00297 
00298    /* copy as a multibyte character string */
00299    DIRENT_STRNCPY ( dirp->curentry.d_name,
00300              dirp->find_data.cFileName,
00301              sizeof(dirp->curentry.d_name) );
00302    dirp->curentry.d_name[MAX_PATH] = '\0';
00303 
00304    /* compute the length of name */
00305    dirp->curentry.d_namlen = strlen (dirp->curentry.d_name);
00306 
00307    /* determine file type */
00308    attr = dirp->find_data.dwFileAttributes;
00309    if ((attr & FILE_ATTRIBUTE_DEVICE) != 0) {
00310       dirp->curentry.d_type = DT_CHR;
00311    } else if ((attr & FILE_ATTRIBUTE_DIRECTORY) != 0) {
00312       dirp->curentry.d_type = DT_DIR;
00313    } else {
00314       dirp->curentry.d_type = DT_REG;
00315    }
00316    return &dirp->curentry;
00317 }

00345 /*****************************************************************************
00346  * Resets the position of the directory stream to which dirp refers to the
00347  * beginning of the directory.  It also causes the directory stream to refer
00348  * to the current state of the corresponding directory, as a call to opendir()
00349  * would have done.  If dirp does not refer to a directory stream, the effect
00350  * is undefined.
00351  */
00352 static void rewinddir(DIR* dirp)
00353 {
00354    if (dirp != NULL) {
00355       /* release search handle */
00356       if (dirp->search_handle != INVALID_HANDLE_VALUE) {
00357          FindClose (dirp->search_handle);
00358       }
00359 
00360       /* open new search handle and retrieve the first entry */
00361       dirp->search_handle = FindFirstFileA (dirp->patt, &dirp->find_data);
00362       if (dirp->search_handle != INVALID_HANDLE_VALUE) {
00363          /* a directory entry is now waiting in memory */
00364          dirp->cached = 1;
00365       } else {
00366          /* failed to re-open directory: no directory entry in memory */
00367          dirp->cached = 0;
00368       }
00369    }
