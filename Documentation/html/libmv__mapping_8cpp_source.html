00057   void MotionFromEssential(const Mat3 &E,
00058     std::vector<Mat3> *Rs,
00059     std::vector<Vec3> *ts) {
00060       Eigen::JacobiSVD<Mat3> USV(E, Eigen::ComputeFullU | Eigen::ComputeFullV);
00061       Mat3 U =  USV.matrixU();
00062       Vec3 d =  USV.singularValues();
00063       Mat3 Vt = USV.matrixV().transpose();
00064 
00065       // Last column of U is undetermined since d = (a a 0).
00066       if (U.determinant() < 0) {
00067         U.col(2) *= -1;
00068       }
00069       // Last row of Vt is undetermined since d = (a a 0).
00070       if (Vt.determinant() < 0) {
00071         Vt.row(2) *= -1;
00072       }
00073 
00074       Mat3 W;
00075       W << 0, -1,  0,
00076         1,  0,  0,
00077         0,  0,  1;
00078 
00079       Mat3 U_W_Vt = U * W * Vt;
00080       Mat3 U_Wt_Vt = U * W.transpose() * Vt;
00081 
00082       Rs->resize(4);
00083       ts->resize(4);
00084       (*Rs)[0] = U_W_Vt;  (*ts)[0] =  U.col(2);
00085       (*Rs)[1] = U_W_Vt;  (*ts)[1] = -U.col(2);
00086       (*Rs)[2] = U_Wt_Vt; (*ts)[2] =  U.col(2);
00087       (*Rs)[3] = U_Wt_Vt; (*ts)[3] = -U.col(2);
00088   }

00241   Mat FivePointsPolynomialConstraints(const Mat &E_basis) {
00242     // Build the polynomial form of E (equation (8) in Stewenius et al. [1])
00243     Vec E[3][3];
00244     for (int i = 0; i < 3; ++i) {
00245       for (int j = 0; j < 3; ++j) {
00246         E[i][j] = Vec::Zero(20);
00247         E[i][j](coef_x) = E_basis(3 * i + j, 0);
00248         E[i][j](coef_y) = E_basis(3 * i + j, 1);
00249         E[i][j](coef_z) = E_basis(3 * i + j, 2);
00250         E[i][j](coef_1) = E_basis(3 * i + j, 3);
00251       }
00252     }
00253 
00254     // The constraint matrix.
00255     Mat M(10, 20);
00256     int mrow = 0;
00257 
00258     // Determinant constraint det(E) = 0; equation (19) of Nister [2].
00259     M.row(mrow++) = o2(o1(E[0][1], E[1][2]) - o1(E[0][2], E[1][1]), E[2][0]) + 
00260       o2(o1(E[0][2], E[1][0]) - o1(E[0][0], E[1][2]), E[2][1]) + 
00261       o2(o1(E[0][0], E[1][1]) - o1(E[0][1], E[1][0]), E[2][2]);
00262 
00263     // Cubic singular values constraint.
00264     // Equation (20).
00265     Vec EET[3][3];
00266     for (int i = 0; i < 3; ++i) {    // Since EET is symmetric, we only compute
00267       for (int j = 0; j < 3; ++j) {  // its upper triangular part.
00268         if (i <= j) {
00269           EET[i][j] = o1(E[i][0], E[j][0])
00270             + o1(E[i][1], E[j][1])
00271             + o1(E[i][2], E[j][2]);
00272         } else {
00273           EET[i][j] = EET[j][i];
00274         }
00275       }
00276     }
00277 
00278     // Equation (21).
00279     Vec (&L)[3][3] = EET;
00280     Vec trace  = 0.5 * (EET[0][0] + EET[1][1] + EET[2][2]);
00281     for (int i = 0; i < 3; ++i) {
00282       L[i][i] -= trace;
00283     }
00284 
00285     // Equation (23).
00286     for (int i = 0; i < 3; ++i) {
00287       for (int j = 0; j < 3; ++j) {
00288         Vec LEij = o2(L[i][0], E[0][j])
00289           + o2(L[i][1], E[1][j])
00290           + o2(L[i][2], E[2][j]);
00291         M.row(mrow++) = LEij;
00292       }
00293     }
00294 
00295     return M;
00296   }

00302     // Gauss Elimination.
00303     for (int i = 0; i < 10; ++i) {
00304       M.row(i) /= M(i,i);

00343       At(8,3) = 1;
00344       At(9,6) = 1;

00347       Eigen::EigenSolver<Mat> es(At);

00349       Matc V = es.eigenvectors();
00350       Matc solutions(4, 10);
00351       solutions.row(0) = V.row(6).array() / V.row(9).array();
00352       solutions.row(1) = V.row(7).array() / V.row(9).array();
00353       solutions.row(2) = V.row(8).array() / V.row(9).array();
00354       solutions.row(3).setOnes();
00355 
00356       // Get the ten candidate E matrices in vector form.
00357       Matc Evec = E_basis * solutions;
00358
