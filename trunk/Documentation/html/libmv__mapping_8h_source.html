00298   double Nullspace(TMat *A, TVec *nullspace) {
00299     Eigen::JacobiSVD<TMat> svd(*A, Eigen::ComputeFullV);
00300     (*nullspace) = svd.matrixV().col(A->cols()-1);
00301     if (A->rows() >= A->cols())
00302       return svd.singularValues()(A->cols()-1);

00304       return 0.0;

00307   inline void EncodeEpipolarEquation(const TMatX &x1, const TMatX &x2, TMatA *A) {
00308     for (int i = 0; i < x1.cols(); ++i) {
00309       (*A)(i, 0) = x2(0, i) * x1(0, i);  // 0 represents x coords,
00310       (*A)(i, 1) = x2(0, i) * x1(1, i);  // 1 represents y coords.
00311       (*A)(i, 2) = x2(0, i);
00312       (*A)(i, 3) = x2(1, i) * x1(0, i);
00313       (*A)(i, 4) = x2(1, i) * x1(1, i);
00314       (*A)(i, 5) = x2(1, i);
00315       (*A)(i, 6) = x1(0, i);
00316       (*A)(i, 7) = x1(1, i);
00317       (*A)(i, 8) = 1.0;
00318     }

00321   // x's are 2D coordinates (x,y,1) in each image; Ps are projective cameras. The
00322   // output, X, is a homogeneous four vectors.
00323   template<typename T>
00324   void NViewTriangulate(const Matrix<T, 2, Dynamic> &x,
00325     const vector<Matrix<T, 3, 4> > &Ps,
00326     Matrix<T, 4, 1> *X) {
00327       int nviews = x.cols();
00328       assert(nviews == Ps.size());
00329 
00330       Matrix<T, Dynamic, Dynamic> design(3*nviews, 4 + nviews);
00331       design.setConstant(0.0);
00332       for (int i = 0; i < nviews; i++) {
00333         design.template block<3, 4>(3*i, 0) = -Ps[i];
00334         design(3*i + 0, 4 + i) = x(0, i);
00335         design(3*i + 1, 4 + i) = x(1, i);
00336         design(3*i + 2, 4 + i) = 1.0;
00337       }
00338       Matrix<T, Dynamic, 1>  X_and_alphas;
00339       Nullspace(&design, &X_and_alphas);
00340       X->resize(4);
00341       *X = X_and_alphas.head(4);
00342   }
