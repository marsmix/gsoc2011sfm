00101         r2 = xn*xn + yn*yn;

00103         // Determine distortion factors (might only work for rational model at this stage)
00104         icdist = (1 + ((radial_dist_[5]*r2 + radial_dist_[4])*r2 + radial_dist_[3])*r2)/
00105           (1 + ((radial_dist_[2]*r2 + radial_dist_[1])*r2 + radial_dist_[0])*r2);
00106         deltaX = 2*tangential_dist_[0]*xn*yn + tangential_dist_[1]*(r2 + 2*xn*xn);
00107         deltaY = tangential_dist_[0]*(r2 + 2*yn*yn) + 2*tangential_dist_[1]*xn*yn;
00108 
00109         // Distort the points, but keep them in normalized co-ordinate system
00110         xc = (xn/icdist) + deltaX;
00111         yc = (yn/icdist) + deltaY;
00112 
00113         redistortedPoints.push_back(cv::Point2d(xc, yc));
