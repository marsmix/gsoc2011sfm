00042   // and the other reconstruction data types is that all points are seen by all

00044   struct NViewDataSet {
00045     std::vector<OpencvSfM::PointOfView> cameras;
00046     std::vector<cv::Vec3d> C;   // Camera centers.
00047     std::vector<cv::Vec3d> X;     // 3D points.
00048     std::vector<std::vector<cv::Vec2d>> x;  // Projected points; may have noise added.
00049     //std::vector<std::vector<int>>  x_ids;// Indexes of points corresponding to the projections
00050 
00051     int n;  // Actual number of cameras.
00052 
00053     cv::Mat P(int i) {
00054       assert(i < n);
00055       return cameras[i].getProjectionMatrix();
00056     }
00057     void Reproject() {
00058       x.clear();
00059       for (int i = 0; i < n; ++i) {
00060         x.push_back(cameras[i].project3DPointsIntoImage(X));
00061       }
00062     }
00063     // TODO(keir): Add gaussian jitter functions.
00064   };
