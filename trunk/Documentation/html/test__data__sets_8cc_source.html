00042   Mat3 LookAt(Vec3 cent) {
00043     Mat projTmp = cent;
00044     Eigen::Map<libmv::Vec3> eigen_tmp((double*)projTmp.data);
00045     Vec3 center = eigen_tmp.transpose();
00046 
00047     Vec3 zc = center.normalized();
00048     Vec3 xc = Vec3::UnitY().cross(zc).normalized();
00049     Vec3 yc = zc.cross(xc);
00050     Mat3 R;
00051     R.row(0) = xc;
00052     R.row(1) = yc;
00053     R.row(2) = zc;
00054     return R;
